<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class User_Input
{
    protected $CI;

    public function __construct()
        {
            $this->CI =& get_instance();
        }

        /**
     * This function is used to return the search term that has been POST'ed when searching the skill table.
     *  If the search term is not stored in a session it will be forgotten so this method
     * stores the term as session data if one does not already exist. This is so the next SELECT LIMIT WHERE
     * query can be ran with this same search term.
     * Returning every record that matches the search term snd storing it would not scale compared to
     * carrying out a SELECT WHERE LIMT query and using pagination to increment the offset of limit. 
     * 
     * @return string $searchterm The searchterm that is either stored from a previous search or newly POST'ed
     */
       public function getSearchTerm()
        {
            $searchterm="";
            $this->CI->load->library('session');
            

            if(!$this->CI->session->userdata('searchterm')){//no search term
                if($this->CI->input->post('search')){   //a new search has been made
                    $searchterm=$this->CI->input->post('search');
                    $this->CI->session->set_userdata('searchterm', $searchterm); //store the new search in session
                }
            } else{ //already a search term
                if($this->CI->input->post('search')){ //new seach has been made
                    $searchterm=$this->CI->input->post('search');
                    $this->CI->session->set_userdata('searchterm', $searchterm); //store the new search in session
                }else{
                    $searchterm=$this->CI->session->userdata('searchterm');  //use previous search term
                }
            }
            
            return $searchterm;
        }
        
}
