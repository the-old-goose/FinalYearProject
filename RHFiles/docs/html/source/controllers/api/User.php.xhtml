<?php

namespace Rhfiles\API;
use Restserver\Libraries\REST_Controller;
use CI_Controller;

defined('BASEPATH') OR exit('No direct script access allowed');

require APPPATH . 'libraries/REST_Controller.php';
require APPPATH . 'libraries/Format.php';

/**
 *
 * This class uses the Rest library which processes HTTP requests returning resources concerning User queries in many formats such as
 * JSON,csv,html,xml by adding optional format parameters to the http request.
 * URI : {api}/{resource}/{parameter}/{format}
 * Credit to Guide: CodeIgniter RestApi tutorial <https://code.tutsplus.com/tutorials/working-with-restful-services-in-codeigniter--net-8814>
 *
 * @author Youcef Adoum
 *
 */
class User extends REST_Controller {
    /**
     * This function is called when a get request is sent to the api controller with parameters of login,
     * The api uses the user model to check for a matching $email and $enc_pass.If successful will return a 
     * JSON object and HTTP_OK (200) response code or HTTP 404 response code
     * A $email and $enc_pass value must be provided in the get request
     * 
     * @return object A JSON object will be returned with a users id
     * @todo login to use POST as GET is unsecure for passwords
     */
    function login_get()
    {   
        $email= $this->get('email');
        $enc_pass=$this->get('pass');

        $user = $this->user_model->login($email,$enc_pass);
    
        if($user)
        {
            $this->response($user, 200); 
        }
 
        else
        {
            $this->response(NULL, 404);//Errors to be handled application side
        }
    }
    
    /**
     * This function is called when a post request is sent to the api controller with a $username,$email and $enc_pass,
     * The api uses the user model to insert the data into the User table
     * An $email and $enc_pass value must be provided in the post request
     * @todo respond with a success response code
     */
    function register_post(){
        $username=$this->input->post('username');
        $email=$this->input->post('email');
        $enc_pass=$this->input->post('password');
        $first_name=$this->input->post('first_name');
        $last_name=$this->input->post('last_name');
        $location_id=$this->input->post('location_id');
        $this->user_model->create_user_and_profile($username,$email,$enc_pass,$first_name,$last_name,$location_id);
    }
}
