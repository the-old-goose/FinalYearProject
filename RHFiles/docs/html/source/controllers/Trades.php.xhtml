<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class Trades extends CI_Controller 
{
    public function initiate($initial_skill_id,$offer_skill_id)
    {
        $user_id=$this->verify_user->checkLoggedIn(); //Verify user logged in, get user_id

        $initial_skill['skill_id']=$initial_skill_id;
        $offer_skill['skill_id']=$offer_skill_id;

        //checks that not own skill proposing for
        if($this->skill_model->read_if_user_created_skill($user_id,$initial_skill_id))
        {
            $this->session->set_flashdata('proposal_error','You Cannot Trade for your own skill');
            redirect('skills/view/'.$initial_skill_id);
            exit();
        }
        //CHECK user owns the offer skill
        if(!$this->skill_model->read_if_user_created_skill($user_id,$offer_skill_id)) 
        {
            $this->session->set_flashdata('proposal_error','You Cannot propose another users skill listing!');
            redirect('skills/view/'.$initial_skill_id);
            exit();
        }
        $sel_date=$this->input->post('sel-date');

        //Checks that a date has been selected using the calendar
        if(empty($_POST['sel-date']))
        {
            $this->session->set_flashdata('proposal_error','You must select atleast 1 day you are available using the calendar!');
            redirect('skills/view/'.$initial_skill_id);
            exit();
        }     
        
        //sel_date array concatenated into a single string seperated by commas and inserted in the db
        $this->trade_model->create_proposal($initial_skill,$offer_skill,implode(",",$sel_date));

        redirect('listings/view_sent_offers');
    }

    public function decide($skill_proposal_id)
    {
        $user_id=$this->verify_user->checkLoggedIn(); //Verify user logged in, get user_id

        //Check if proposal is not pending approval
        if(!$this->trade_model->read_status($skill_proposal_id,TRUE,"pending approval"))
        {
            $this->session->set_flashdata('decide_error','This trade is no longer pending approval');
            redirect("listings/view_offers");
            exit();
        }
        
        //Get details of proposal
        $data['proposal']=$this->trade_model->read_proposal($user_id,$skill_proposal_id);

        //TODO:Ensure user_id has created the initial skill

        empty($data['proposal']) ? $data['noresults']=TRUE: $data['noresults']=FALSE;
        
        //Read proposed dates
        $proposed_dates[]=$this->proposal_model->read_proposal_dates($skill_proposal_id);

        //Generate Calendar with proposed dates marked with checkboxes in view
        $this->load->library('availability', $proposed_dates);
        $data['calendar']=$this->availability->generate_calendar();
        
        $this->load->view('templates/header', $data);
        $this->load->view('trades/view',$data);
        $this->load->view('templates/footer', $data);
    }

    public function view_progress($skill_proposal_id=FALSE)
    {
        $user_id=$this->verify_user->checkLoggedIn(); //Verify user logged in, get user_id
        
        if(!$skill_proposal_id) //No proposal_id given
        {
            redirect('listings/view_offers');
            exit();
        }
     
        $data['status']=$this->trade_model->read_status($skill_proposal_id);

        if($data['status']=='inprogress') // If the trade is 'inprogress' read the other users name and accepted date of trade
        {
          
           $data['proposal']=$this->trade_model->read_accepted_date_and_opposite_username($user_id,$skill_proposal_id);
        }

        $data['skill_proposal_id']=$skill_proposal_id;

        $this->load->view('templates/header', $data);
        $this->load->view('trades/progress',$data);
        $this->load->view('templates/footer', $data);  
    }

    public function view_history()
    {
        $user_id=$this->verify_user->checkLoggedIn(); //Verify user logged in, get user_id

        //Read trade history
       
        $data['trades']=$this->trade_model->read_trade_history($user_id);

        empty($data['trades']) ? $data['no_results']=true: $data['no_results']=false;

        $this->load->view('templates/header', $data);
        $this->load->view('listings/history',$data);
        $this->load->view('templates/footer', $data);  
    }

    public function accept_proposal($skill_proposal_id)
    {
        $user_id=$this->verify_user->checkLoggedIn(); //Verify user logged in, get user_id

        //Catch No date Selected

        $accepted_date=$this->input->post('avail_selected');

        if($accepted_date<date('j')) //date selected less than day of month it is
        {
            $accepted_date=date('M',strtotime('+1 month')).$accepted_date; //append a month prefix e.g Jan onto the date
        }
        else //else append the current months prefix e.g Dec
        {
            $accepted_date=date('M').$accepted_date;
        }

        $this->trade_model->start_trade($skill_proposal_id,$accepted_date);
        redirect('trades/view_progress/'.$skill_proposal_id); 
    }

    public function cancel_proposal($skill_proposal_id)
    {
        $user_id=$this->verify_user->checkLoggedIn(); //Verify user logged in, get user_id

        //TODO:check right user can decline the proposal

        $this->trade_model->decline_trade($skill_proposal_id);

        $this->load->view('templates/header', $data);
        $this->load->view('listings/view_offers',$data);
        $this->load->view('templates/footer', $data);  
    }

    public function complete_trade($skill_proposal_id)
    {
        $user_id=$this->verify_user->checkLoggedIn(); //Verify user logged in, get user_id

        

        //verify its in progress
        $this->trade_model->create_trade($skill_proposal_id); 

        //TODO:Add score to both users

        redirect('trades/view_progress/'.$skill_proposal_id);

    }

}
