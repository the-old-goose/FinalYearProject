     * This function is called when the user attempts to register. The fields sent by the POST request are
     * validated using codeigniters validation library and one custom check defined in this class.
     * If validation is failed reload the page and display the errors, if it passes then encrypt the password
     * and send all the data in another post request to the api which will respond with a success or failure.
     * Then redirect to the homepage setting codeigniter flash data, notifying the user which will apear once.
     * 
     * @todo Implement cURL

    public function register()
    {
            if(isset($_SESSION['user_id'])) //Already logged in?
            {
                redirect('profiles/view');
                exit();
            }

            if($this->verify_user->verify_user_register()===FALSE) //validation failed a field empty or username exists 
            {
                $this->load->view('templates/header');
                $this->load->view('users/register');//reload the page with validation error msgs
                $this->load->view('templates/footer');
            }
            else //POST user info to api  and redirect back to homepage
            { 
                $url = 'http://localhost/poc2/index.php/api/user/register';
                $encrypt_pass=sha1($this->input->post('pass'));

                $data= array(
                    'user_id'=>NULL,
                    'username'=>$this->input->post('user'),
                    'email'=>$this->input->post('email'),
                    'password'=>$encrypt_pass,
                    'user_type'=>1, //1 set user_type as a basic user as default when registering.
                    'signup_date'=>NULL,
                    'location_id'=>$this->input->post('locationid'),
                    'first_name'=>ucfirst($this->input->post('first_name')),
                    'last_name'=>ucfirst($this->input->post('last_name'))
                );

                $options = array(
                    'http' => array(
                        'header'  => "Content-type: application/x-www-form-urlencoded\r\n",
                        'method'  => 'POST',
                        'content' => http_build_query($data)
                    )
                );

                $context  = stream_context_create($options); //make POST request to api
                $result = file_get_contents($url, false, $context);
                
                if ($result === FALSE) 
                { 
                    //todo: hide failure and display errors in view
                    echo "<h1> Register UNSUCCESSFUL!</h1>";
                 }
               
                $this->session->set_flashdata('succ_msg','You are registered successfully');
                redirect('users/login');
            }
        }

        /**
         * This function is called when logging out, it will unset the session data.The user is 
         * redirected to the homepage and flash data is displayed
         */
        public function logout(){
            $this->session->set_flashdata('succ_msg','You are now logged out');

            $this->session->unset_userdata('logged_in');
            $this->session->unset_userdata('user_id');
            $this->session->unset_userdata('username');
            redirect('users/login');
        }
