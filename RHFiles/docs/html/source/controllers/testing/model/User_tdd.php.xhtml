class User_tdd extends CI_Controller
{

        /***** TEST CASE *****/

        $test4=$this->cleanupRegisteredAccounts($username,$lastname);

    }

    /**
     * This test checks that the test case does not already exist prior to testing
     */
    private function checkingUserDoesNotExist($username)
    {
        return !$success =$this->user_model->check_exists($username);
    }
    
    /**
     * This test checks that creating a user with valid details returns true when inserting the data
     * into the database.
     */
    private function registrationWithValidDetails($fname,$lname,$username,$email,$enc_pass)
    {
        return  $success= $this->user_model->create_user_and_profile($username,$email,$enc_pass,$fname,$lname); 
    }

    /**
     * This test checks that creating a user with valid details then allows the same user to log in via
     * the user model.
     */
    private function registrationThenCheckingUserLogin($fname,$lname,$username,$email,$enc_pass)
    {
        $this->user_model->create_user_and_profile($username,$email,$enc_pass,$fname,$lname);
        return $success=$this->user_model->login($email,$enc_pass);
    }

    private function assertThatProfileExists($firstname,$lastname,$username,$email,$pass){
        $user_id=$this->user_model->create_user_and_profile($username,$email,$pass,$firstname,$lastname,$test_flag=TRUE);
        $profile=$this->profile_model->read_profile($user_id);
        return ($profile['first_name']==$firstname && $profile['last_name']==$lastname );
    }

    /**
     * This function removes the account created in the test run to allow test re-runs.
     */
    private function cleanupRegisteredAccounts($username,$last_name)
    {
        return $this->user_model->delete_user($username,$last_name);
    }

}
