<?php
defined('BASEPATH') OR exit('No direct script access allowed');

/**
 * 
 * This Class extends the CI_controller and is used to display static pages in the site.
 * The page as a variable is passed in the URL and the corrisponding view is routed
 * and displayed to the user.
 * 
 * @see <https://www.codeigniter.com/userguide3/tutorial/static_pages.html>
 *
 */
class Pages extends CI_Controller {
    
    /**
     * 
     * This function will load the corrisponding view along with the page header and footer.
     * If the $page provided does not exist in the views folder, 404 error will be displayed.
     * 
     * @param string $page Name of a page in the site.
     */
    public function view($page = 'home') {
        if ( !file_exists(APPPATH.'views/pages/'.$page.'.php'))
        {
            show_404();
        }

        if($page =='home')
        {
            $this->load->view('templates/head');
            $this->load->view('pages/home');
            
        }
        else
        {
            $this->load->view('templates/header');
            $this->load->view('pages/'.$page);
            $this->load->view('templates/footer');
        }
    }
}
